#how to setup your own branch and push changes to the branch

git branch <branch_name>
#ex: git branch nkdev

#to change into the newly created branch do:
git checkout <branch_name>

#this next command somehow makes the branch visible on the online repo
git push --set-upstream origin <branch_name>

#after all of this is setup, here's how to update the files on your branch:

git add <file_that_has_been_editied or new_file>

git commit -m <"short message detailing the changes to the file">

git push
